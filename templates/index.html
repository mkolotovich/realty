Как добавить новый сайт

01
Создайте файл парсера
# parsers/my_site_parser.py
from .base import BaseParser

class MySiteParser(BaseParser):
    name         = "my_site"
    display_name = "My Site"
    base_url     = "https://my-site.ru"

    def parse(self, filters):
        soup = self.get_page(self.base_url + "/search", params={
            "city": filters.get("city"),
            "deal": filters.get("deal_type"),
        })
        listings = []
        for card in soup.select(".card"):
            listings.append(self._listing(
                title   = card.select_one(".title").text,
                price   = int(card.select_one(".price").text.replace(" ", "")),
                address = card.select_one(".addr").text,
                url     = card.select_one("a")["href"],
            ))
        return listings
02
Зарегистрируйте парсер
# parsers/registry.py
from .my_site_parser import MySiteParser

PARSERS = {
    "demo": DemoParser(),
    "my_site": MySiteParser(),   # ← добавьте
}
03
Перезапустите сервер
python app.py

Новый источник появится в списке фильтров автоматически.
